summernote文本编辑器的使用

介绍

summernote是一款Jquery文本编辑器的插件，本文只是简单的实现文本编辑及图片上传下载功能，更多api请参考summernote官方文档

引入

summernote依赖于bootstrap和jquery所以也需要引入bootstrap和jquery

<link rel="stylesheet" type="text/css" href="/assets/css/lib/bootstrap.min.css" >
<link rel="stylesheet" type="text/css" href="/css/summernote.css">
<script src="/assets/js/lib/jquery.min.js"></script>
<script src="/assets/js/lib/bootstrap.min.js"></script>
<script src="/js/summernote.js"></script>

<div id="summernote"></div>

js

// summernote的初始化方法
$("#summernote").summernote({
    placeholder: "输入内容",
    tabsize: 2,
    height: 300,
    lang: 'zh-CN',
    focus: true,
    callbacks: {
        onImageUpload: function (files) {
        		// 上传图片
            uploadFile(files[0]);
        },
        onMediaDelete: function (target){
        		// 删除图片
            deleteFile(target);
        }
    }
});

// 上传文件
function uploadFile(file){
    var imgPath = sendFile(file);
    $('#summernote').summernote('insertImage', imgPath);
}

// 用于summernote内容回显
function getContent(){
    var content = $('#summernote').summernote('code');
    if (content=="<p><br></p>"){
        return '';
    }
    return content;
}

// 获取summernote的内容
function addContent(content){
    $('#summernote').summernote('code', content);
}

// 上传图片至服务器
function sendFile(file) {
    if (file == null||file==undefined) {
        return null;
    }
    var formData = new FormData();
    formData.append("file", file);
    console.log(formData);
    var filePath = "";
    $.ajax({
        url: "/file/upload",
        type: "POST",
        data: formData,
        async: false,
        contentType: false,
        processData: false,
        success: function (result) {
            filePath = result.data.filePath;
        }
    });
    return filePath;
}

// 删除服务器图片
function deleteFile(filePath) {
    $.ajax({
        url: "/file/delete",
        type: "POST",
        data: {
            "filePath": filePath
        },
        success: function (result) {
            console.log(result);
        }
    })
}

java接口

@RequestMapping("/addImg")
public Map addImg(@RequestParam("file") MultipartFile file, HttpServletRequest request) throws IOException {
    String fileName = file.getOriginalFilename();
    fileName.split(".");
    System.out.println("接收到文件:" + fileName);
    String fileType = fileName.substring(fileName.lastIndexOf("."));
    String filePath = new File("").getCanonicalFile().getPath() + "/static/images/";
    filePath = filePath + UUID.randomUUID().toString() + fileType;
    File dest = new File(filePath);
    if (dest.getParentFile() != null) {
        dest.getParentFile().mkdirs();
    }
    try {
        file.transferTo(dest);
    } catch (IOException e) {
        log.warning("文件上传失败:" + e);
        return ResultMap.failed("文件上传失败");
    }
    String path = request.getScheme() + "://" + request.getServerName() + ":"
            + request.getServerPort() + request.getContextPath() + "/images/" + dest.getName();
    return ResultMap.success(path);
}

@RequestMapping("deleteImg")
public Map deleteImg(String imgSrc) {
    String fileName = imgSrc.substring(imgSrc.lastIndexOf("/") + 1);
    System.out.println(fileName);
    String filePath = Thread.currentThread().getContextClassLoader().getResource("").getPath() + "static/images/";
    filePath = filePath + fileName;
    File dest = new File(filePath);
    return dest.delete() ? ResultMap.success() : ResultMap.failed();
}
